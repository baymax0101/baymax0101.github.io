数据类型

核心概念: Oracle 有丰富的数据类型，包括数值、字符串、日期等，每种类型有不同的存储方式和使用场景。

数值类型

-- NUMBER: 数值类型
id NUMBER(5)          -- 5位整数，范围 -99999 到 99999
height NUMBER(3,2)   -- 共3位，2位小数（如 12.34）
price NUMBER         -- 变长数值
balance NUMBER(10,2) -- 金额类型

字符串类型

核心概念: 字符串分定长（CHAR）和不定长（VARCHAR2），建议用 VARCHAR2。

-- VARCHAR2: 可变长度字符串（推荐使用）
name VARCHAR2(50)        -- 最大50字符
description VARCHAR2(2000)

-- CHAR: 定长字符串（不足部分用空格填充）
gender CHAR(1)           -- 固定1个字符
code CHAR(10)            -- 固定10个字符

-- CLOB: 大型文本（最大4GB，推荐使用）
article CLOB            -- 大文本
resume CLOB

-- 不推荐使用 LONG（Oracle 早期类型，有诸多限制）
long_col LONG           -- 最大2GB，不能建索引，不能作为WHERE条件

字符串函数

核心概念: 处理文本数据，包括连接、截取、替换、格式化等。

-- CONCAT: 连接两个字符串
SELECT CONCAT('Hello', 'World') FROM DUAL;  -- HelloWorld

-- || : 字符串连接（多个）
SELECT 'Hello' || ' ' || 'World' FROM DUAL;  -- Hello World

-- LENGTH: 返回字符串长度
SELECT LENGTH('hello') FROM DUAL;  -- 5
SELECT LENGTH('你好') FROM DUAL;   -- 2（字符数）

-- LENGTHB: 返回字节长度
SELECT LENGTHB('你好') FROM DUAL;  -- 4（UTF8一个汉字占3字节）

-- UPPER/LOWER/INITCAP: 大小写转换
SELECT UPPER('hello') FROM DUAL;         -- HELLO
SELECT LOWER('HELLO') FROM DUAL;         -- hello
SELECT INITCAP('hello world') FROM DUAL; -- Hello World

-- TRIM/LTRIM/RTRIM: 去除空格或指定字符
SELECT TRIM('  hello  ') FROM DUAL;      -- hello
SELECT TRIM('A' FROM 'AABCCAA') FROM DUAL; -- B BCC
SELECT LTRIM('xxhelloxx', 'x') FROM DUAL; -- helloxx
SELECT RTRIM('xxhelloxx', 'x') FROM DUAL; -- xxhello

-- LPAD/RPAD: 左/右填充
SELECT LPAD('ABC', 10, '*') FROM DUAL;  -- *******ABC
SELECT RPAD('ABC', 10, '*') FROM DUAL;  -- ABC*******

-- SUBSTR: 截取子字符串
SELECT SUBSTR('Hello World', 1, 5) FROM DUAL;  -- Hello（从1开始）
SELECT SUBSTR('Hello World', 7) FROM DUAL;     -- World
SELECT SUBSTR('Hello World', -5) FROM DUAL;    -- World（从右边）

-- INSTR: 查找子串位置
SELECT INSTR('Hello World', 'o') FROM DUAL;          -- 5
SELECT INSTR('Hello World', 'o', 6) FROM DUAL;        -- 8（从第6位开始）
SELECT INSTR('Hello World', 'o', 1, 2) FROM DUAL;     -- 8（第2次出现）

-- REPLACE: 替换字符串
SELECT REPLACE('Hello World', 'World', 'Oracle') FROM DUAL; -- Hello Oracle

-- TRANSLATE: 字符逐个映射
SELECT TRANSLATE('A1B2', '12', 'XY') FROM DUAL;  -- AXBY

-- ASCII/CHR: ASCII码与字符转换
SELECT ASCII('A') FROM DUAL;   -- 65
SELECT CHR(65) FROM DUAL;      -- A

日期类型

核心概念: DATE 是 Oracle 最常用的日期类型，存储精确到秒；TIMESTAMP 更精确。

-- DATE: 日期时间（精确到秒）
birth DATE              -- 格式: DD-MON-YY，如 25-3月-24

-- TIMESTAMP: 时间戳（精确到毫秒）
create_time TIMESTAMP   -- 格式: 25-MAR-24 10.30.45.123456 AM

-- TIMESTAMP WITH TIME ZONE: 带时区
update_time TIMESTAMP WITH TIME ZONE

-- TIMESTAMP WITH LOCAL TIME ZONE: 本地时区
local_time TIMESTAMP WITH LOCAL TIME ZONE

-- 日期类型示例
INSERT INTO employees (hire_date) VALUES (TO_DATE('2024-01-01', 'YYYY-MM-DD'));
INSERT INTO employees (hire_date) VALUES (TO_DATE('2024-01-01 10:30:00', 'YYYY-MM-DD HH24:MI:SS'));

-- SYSDATE / SYSTIMESTAMP: 系统当前时间
SELECT SYSDATE FROM DUAL;        -- 当前日期时间
SELECT SYSTIMESTAMP FROM DUAL;   -- 带毫秒和时区

数据库基础

核心概念: Oracle 是一个商业级关系数据库，用表空间(Tablespace)存放数据，用户管理权限。

-- 注释用两个连字符，PL/SQL块用 / 结束

-- 创建表空间（存放数据的仓库）
CREATE TABLESPACE mytablespace 
DATAFILE 'mytablespace.dbf' SIZE 100M 
AUTOEXTEND ON NEXT 10M MAXSIZE UNLIMITED;

-- 删除表空间
DROP TABLESPACE mytablespace INCLUDING CONTENTS AND DATAFILES;

-- 创建用户（Oracle中用户即 schema）
CREATE USER identified BY password 
DEFAULT TABLESPACE mytablespace
TEMPORARY TABLESPACE temp;

-- 授权（Oracle开发必备权限）
GRANT CONNECT, RESOURCE TO myuser;
GRANT DBA TO myuser;  -- 管理员权限

-- 登录
-- CONNECT myuser/password

-- 列出当前用户下的表
SELECT table_name FROM user_tables;

-- 列出当前用户可访问的表
SELECT owner, table_name FROM all_tables WHERE owner = 'OTHER_USER';

表操作

DDL:数据定义语言，包括CREATE ALTER DROP TURNCATE

DML:数据操作语言，包括INSERT UPDATE DELETE

核心概念: 创建、修改、删除表，以及数据的增删改查。Oracle 用 VARCHAR2 而非 VARCHAR。

-- 创建表（Oracle 用 VARCHAR2）
CREATE TABLE employees (
    emp_id NUMBER(10) PRIMARY KEY,
    emp_name VARCHAR2(100) NOT NULL,
    email VARCHAR2(100) UNIQUE,
    hire_date DATE DEFAULT SYSDATE,
    salary NUMBER(10, 2),
    dept_id NUMBER(10)
);

-- 修改表结构
-- 修改表名
RENAME TBALE employees TO emp;
-- 添加列,只能添加到末尾
ALTER TABLE employees ADD (phone VARCHAR2(20));

-- 修改列
ALTER TABLE employees MODIFY (emp_name VARCHAR2(200));

-- 修改列名
ALTER TABLE employees RENAME COLUMN emp_name TO name;

-- 删除列
ALTER TABLE employees DROP COLUMN phone;

-- 删除表
DROP TABLE employees PURGE;  -- PURGE 彻底删除，不放回收站

-- 插入数据(指定列)，可以不用指定所有列，但是非空约束的字段必须赋值
INSERT INTO employees (emp_id, emp_name, email, salary, dept_id)
VALUES (1, '张三', 'zhangsan@example.com', 8000, 10);

-- 默认格式
INSERT INTO emplpoyees VALUES (2,'赵四','zhaosi@gamil.com',7000,10);
-- 批量插入
INSERT INTO employees (emp_id, emp_name, salary)
VALUES 
    (2, '李四', 7000),
    (3, '王五', 9000),
    (4, '赵六', 8500);

-- 更新数据，需要搭配WHERE使用
UPDATE employees SET salary = 10000 WHERE emp_id = 1;

-- 删除数据，from可写可不写，不加where会删掉整张表
DELETE FROM employees WHERE emp_id = 1;

-- 清空表（更快，类似 delete），无法回滚
TRUNCATE TABLE employees;

-- 以上操作只是在自己的视图中完成，并没有同步到数据库，同步需要TCL（事务控制语言）
-- 提交（Oracle 默认不自动提交）
COMMIT;
-- 回滚,前提是还没有进行commit操作
ROLLBACK;

约束

核心概念: 约束是数据库中保证数据完整性的规则，就像”规则检查器”，不符合条件的数据无法插入或更新。

约束类型

-- 约束类型说明
-- NOT NULL    : 非空约束，字段必须有值
-- UNIQUE      : 唯一约束，字段值不能重复
-- PRIMARY KEY : 主键约束，非空且唯一，标识每条记录
-- FOREIGN KEY : 外键约束，引用其他表的数据
-- CHECK       : 检查约束，限制字段取值范围

-- 1. 创建表时添加约束

-- 方式一：列级约束（直接在列定义后面）
CREATE TABLE employees (
    emp_id NUMBER(10) PRIMARY KEY,                    -- 主键
    emp_name VARCHAR2(100) NOT NULL,                   -- 非空
    email VARCHAR2(100) UNIQUE,                        -- 唯一
    phone VARCHAR2(20) NOT NULL,                      -- 非空
    salary NUMBER(10, 2) CHECK (salary > 0),           -- 检查约束
    dept_id NUMBER(10) REFERENCES departments(dept_id) -- 外键
);

-- 方式二：表级约束（在所有列定义后单独定义）
CREATE TABLE employees (
    emp_id NUMBER(10),
    emp_name VARCHAR2(100) NOT NULL,
    email VARCHAR2(100),
    salary NUMBER(10, 2),
    dept_id NUMBER(10),
    -- 主键约束（表级）
    CONSTRAINT pk_emp PRIMARY KEY (emp_id),
    -- 唯一约束
    CONSTRAINT uk_email UNIQUE (email),
    -- 检查约束
    CONSTRAINT ck_salary CHECK (salary > 0),
    -- 外键约束
    CONSTRAINT fk_dept FOREIGN KEY (dept_id) REFERENCES departments(dept_id)
);

-- 复合主键（多个列共同作为主键）
CREATE TABLE order_items (
    order_id NUMBER(10),
    product_id NUMBER(10),
    quantity NUMBER(5),
    CONSTRAINT pk_order_item PRIMARY KEY (order_id, product_id)
);

-- 2. 修改表添加约束

-- 添加主键
ALTER TABLE employees ADD CONSTRAINT pk_emp PRIMARY KEY (emp_id);

-- 添加唯一约束
ALTER TABLE employees ADD CONSTRAINT uk_email UNIQUE (email);

-- 添加非空约束（使用 MODIFY）
ALTER TABLE employees MODIFY (emp_name NOT NULL);

-- 添加外键约束
ALTER TABLE employees ADD CONSTRAINT fk_dept 
    FOREIGN KEY (dept_id) REFERENCES departments(dept_id);

-- 添加检查约束
ALTER TABLE employees ADD CONSTRAINT ck_salary 
    CHECK (salary > 0 AND salary < 100000);

-- 3. 约束操作

-- 查看表的约束
SELECT constraint_name, constraint_type, status
FROM user_constraints 
WHERE table_name = 'EMPLOYEES';

-- 查看约束对应的列
SELECT constraint_name, column_name
FROM user_cons_columns
WHERE table_name = 'EMPLOYEES';

-- 禁用约束（数据导入时常用）
ALTER TABLE employees DISABLE CONSTRAINT pk_emp;
ALTER TABLE employees DISABLE CONSTRAINT fk_dept;

-- 启用约束
ALTER TABLE employees ENABLE CONSTRAINT pk_emp;
ALTER TABLE employees ENABLE CONSTRAINT fk_dept;

-- 删除约束
ALTER TABLE employees DROP CONSTRAINT pk_emp;
ALTER TABLE employees DROP PRIMARY KEY;  -- 删除主键（简写）

-- 延迟约束（事务提交时才检查）
-- 创建时指定
CREATE TABLE orders (
    order_id NUMBER(10) PRIMARY KEY,
    amount NUMBER(10, 2),
    CONSTRAINT ck_amount CHECK (amount > 0) DEFERRABLE INITIALLY DEFERRED
);

-- 运行时修改
SET CONSTRAINT ck_amount DEFERRED;  -- 延迟检查
SET CONSTRAINT ck_amount IMMEDIATE; -- 立即检查

约束示例

-- 示例表：员工表 employees
CREATE TABLE employees (
    emp_id NUMBER(10) CONSTRAINT pk_emp PRIMARY KEY,
    emp_name VARCHAR2(100) CONSTRAINT nn_emp_name NOT NULL,
    email VARCHAR2(100) CONSTRAINT uk_email UNIQUE,
    phone VARCHAR2(20),
    salary NUMBER(10, 2) CONSTRAINT ck_salary CHECK (salary >= 0),
    commission_pct NUMBER(5, 2),
    hire_date DATE DEFAULT SYSDATE,
    manager_id NUMBER(10),
    dept_id NUMBER(10) CONSTRAINT fk_dept REFERENCES departments(dept_id),
    CONSTRAINT ck_commission CHECK (commission_pct >= 0 AND commission_pct <= 1)
);

-- 示例表：部门表 departments
CREATE TABLE departments (
    dept_id NUMBER(10) CONSTRAINT pk_dept PRIMARY KEY,
    dept_name VARCHAR2(100) CONSTRAINT nn_dept_name NOT NULL,
    manager_id NUMBER(10),
    location_id NUMBER(10)
);

-- 示例表：职位表 jobs
CREATE TABLE jobs (
    job_id VARCHAR2(10) CONSTRAINT pk_job PRIMARY KEY,
    job_title VARCHAR2(100) CONSTRAINT nn_job_title NOT NULL,
    min_salary NUMBER(10, 2),
    max_salary NUMBER(10, 2),
    CONSTRAINT ck_salary_range CHECK (min_salary < max_salary)
);

-- 自引用外键（员工管理关系）
ALTER TABLE employees ADD CONSTRAINT fk_manager 
    FOREIGN KEY (manager_id) REFERENCES employees(emp_id);

-- ON DELETE 级联（删除部门时级联删除员工）
ALTER TABLE employees 
ADD CONSTRAINT fk_dept_cascade 
FOREIGN KEY (dept_id) REFERENCES departments(dept_id) ON DELETE CASCADE;

-- ON DELETE SET NULL（删除部门时员工部门设为NULL）
ALTER TABLE employees 
ADD CONSTRAINT fk_dept_setnull 
FOREIGN KEY (dept_id) REFERENCES departments(dept_id) ON DELETE SET NULL;

-- 约束组合示例
-- 联合唯一约束（同一部门不能有同名员工）
CREATE TABLE employees (
    dept_id NUMBER(10),
    emp_name VARCHAR2(100),
    CONSTRAINT uk_dept_emp UNIQUE (dept_id, emp_name)
);

-- 复杂检查约束
ALTER TABLE employees ADD CONSTRAINT ck_emp_info 
CHECK (
    (salary > 0 AND commission_pct IS NOT NULL) OR 
    (salary > 0 AND commission_pct IS NULL) OR
    (salary IS NULL)
);

-- 约束验证
-- ENABLE NOVALIDATE: 允许旧数据不符合，但新数据必须符合
ALTER TABLE employees ENABLE NOVALIDATE CONSTRAINT ck_salary;

约束相关查询

-- 查询当前用户所有约束
SELECT constraint_name, constraint_type, table_name, status
FROM user_constraints
ORDER BY table_name;

-- 约束类型说明
-- C = CHECK
-- P = PRIMARY KEY
-- R = FOREIGN KEY
-- U = UNIQUE

-- 查询约束对应的列
SELECT a.constraint_name, a.column_name, b.constraint_type
FROM user_cons_columns a
JOIN user_constraints b ON a.constraint_name = b.constraint_name
WHERE a.table_name = 'EMPLOYEES'
ORDER BY a.position;

-- 查询外键引用关系
SELECT a.table_name AS 子表,
       a.constraint_name AS 外键约束,
       b.table_name AS 父表,
       c.column_name AS 外键列,
       d.column_name AS 引用列
FROM user_constraints a
JOIN user_constraints b ON a.r_constraint_name = b.constraint_name
JOIN user_cons_columns c ON a.constraint_name = c.constraint_name
JOIN user_cons_columns d ON b.constraint_name = d.constraint_name AND c.position = d.position
WHERE a.constraint_type = 'R';

-- 查询哪些表引用了某表
SELECT a.table_name AS referencing_table,
       a.constraint_name
FROM user_constraints a
WHERE a.r_constraint_name IN (
    SELECT constraint_name 
    FROM user_constraints 
    WHERE table_name = 'DEPARTMENTS' AND constraint_type IN ('P', 'U')
);

-- 查看约束是否启用
SELECT constraint_name, table_name, status, validated
FROM user_constraints
WHERE table_name = 'EMPLOYEES';

-- 查看禁用状态的约束
SELECT constraint_name, table_name, status
FROM user_constraints
WHERE status = 'DISABLED';

基础查询

核心概念: 查询是 SQL 的核心，Oracle 用 ROWNUM 或 FETCH 实现分页。

-- 查询所有列
SELECT * FROM employees;

-- 查询指定列
SELECT emp_id, emp_name, salary FROM employees;

-- 条件查询
SELECT * FROM employees WHERE salary > 8000;

-- 给列起别名
-- 在as关键字后跟别名，如过别名区分大小写或者带空格，必须写在双引号中
SELECT ename as name,sal*12 "annual salary" FROM employees;
-- select 后加上 distinct 可以去除重复行

-- LIKE 模糊匹配
SELECT * FROM employees WHERE emp_name LIKE '%张%';
SELECT * FROM employees WHERE emp_name LIKE '张_';  -- _代表单个字符

-- 排序（默认升序 ASC，降序用 DESC）
SELECT * FROM employees ORDER BY salary DESC, emp_name ASC;

-- 分页查询（Oracle 12c+ 可用 FETCH）
-- 方法1: ROWNUM（经典方式），伪列，返回标识行数据顺序的数字（添加行号）
-- 查询一行添加一个行号，从1开始递增
-- 下面使用了行内视图，rownum可以作为结果集中的数据查询
SELECT * FROM (
    SELECT ROWNUM rn, emp_id, emp_name, salary 
    FROM employees WHERE ROWNUM <= 10
) WHERE rn > 0;

-- 方法2: OFFSET（Oracle 12c+）
SELECT emp_id, emp_name, salary 
FROM employees 
ORDER BY emp_id 
OFFSET 0 ROWS FETCH NEXT 10 ROWS ONLY;

-- 去重
SELECT DISTINCT dept_id FROM employees;

-- 聚合函数
SELECT 
    COUNT(*) AS 总人数,
    AVG(salary) AS 平均工资,
    MAX(salary) AS 最高工资,
    MIN(salary) AS 最低工资,
    SUM(salary) AS 工资总和
FROM employees;

-- 分组
SELECT dept_id, AVG(salary) AS avg_salary 
FROM employees 
GROUP BY dept_id;

-- 分组后过滤（用 HAVING）
SELECT dept_id, AVG(salary) AS avg_salary 
FROM employees 
GROUP BY dept_id 
HAVING AVG(salary) > 7000;

-- CASE 表达式（相当于 if-else）
SELECT emp_name, salary,
    CASE 
        WHEN salary < 5000 THEN '低薪'
        WHEN salary < 10000 THEN '中等'
        ELSE '高薪'
    END AS 工资等级
FROM employees;

-- NVL/COALESCE 处理空值
SELECT emp_name, NVL(commission, 0) FROM employees;
SELECT COALESCE(a, b, c) FROM table1;  -- 返回第一个非空值

Oracle函数

核心概念: Oracle 提供丰富的内置函数，用于数据处理、转换和分析。

日期函数

核心概念: 日期是数据库中非常重要的类型，Oracle 有强大的日期处理能力。

-- 常用日期函数
-- SYSDATE: 当前日期时间（精确到秒）
SELECT SYSDATE FROM DUAL;

-- SYSTIMESTAMP: 当前日期时间（精确到毫秒+时区）
SELECT SYSTIMESTAMP FROM DUAL;

-- CURRENT_DATE / CURRENT_TIMESTAMP: 当前会话时区的日期时间
SELECT CURRENT_DATE, CURRENT_TIMESTAMP FROM DUAL;

-- ADD_MONTHS: 加减月份
SELECT ADD_MONTHS(SYSDATE, 3) FROM DUAL;       -- 3个月后
SELECT ADD_MONTHS(SYSDATE, -6) FROM DUAL;      -- 6个月前

-- MONTHS_BETWEEN: 计算两个日期相差的月份
SELECT MONTHS_BETWEEN(SYSDATE, hire_date) FROM employees;

-- LAST_DAY: 返回指定日期所在月的最后一天
SELECT LAST_DAY(SYSDATE) FROM DUAL;

-- NEXT_DAY: 返回指定日期之后的下一个星期几
SELECT NEXT_DAY(SYSDATE, '星期一') FROM DUAL;  -- 下一个周一
SELECT NEXT_DAY(SYSDATE, 1) FROM DUAL;         -- 1=周日，2=周一...

-- ROUND / TRUNC: 日期四舍五入/截断
SELECT ROUND(SYSDATE, 'YEAR') FROM DUAL;   -- 四舍五入到年
SELECT ROUND(SYSDATE, 'MONTH') FROM DUAL;  -- 四舍五入到月
SELECT ROUND(SYSDATE, 'DD') FROM DUAL;     -- 四舍五入到天
SELECT TRUNC(SYSDATE, 'YEAR') FROM DUAL;   -- 截断到年的第一天
SELECT TRUNC(SYSDATE, 'MONTH') FROM DUAL;  -- 截断到月的第一天
SELECT TRUNC(SYSDATE) FROM DUAL;           -- 截断到天（00:00:00）

-- EXTRACT: 提取日期组成部分
SELECT EXTRACT(YEAR FROM SYSDATE) FROM DUAL;
SELECT EXTRACT(MONTH FROM SYSDATE) FROM DUAL;
SELECT EXTRACT(DAY FROM SYSDATE) FROM DUAL;
SELECT EXTRACT(HOUR FROM SYSTIMESTAMP) FROM DUAL;
SELECT EXTRACT(MINUTE FROM SYSTIMESTAMP) FROM DUAL;

-- 日期运算
SELECT SYSDATE + 1 FROM DUAL;        -- 加1天
SELECT SYSDATE - 1 FROM DUAL;        -- 减1天
SELECT SYSDATE + 1/24 FROM DUAL;     -- 加1小时
SELECT SYSDATE + 1/24/60 FROM DUAL;  -- 加1分钟
SELECT SYSDATE + 7 FROM DUAL;         -- 一周后

-- 转换函数
-- TO_CHAR: 日期转字符串
SELECT TO_CHAR(SYSDATE, 'YYYY-MM-DD') FROM DUAL;
SELECT TO_CHAR(SYSDATE, 'YYYY-MM-DD HH24:MI:SS') FROM DUAL;
SELECT TO_CHAR(SYSDATE, 'YYYY"年"MM"月"DD"日"') FROM DUAL;
SELECT TO_CHAR(SYSDATE, 'DAY') FROM DUAL;        -- 星期几
SELECT TO_CHAR(SYSDATE, 'DY') FROM DUAL;         -- 星期几（简写）
SELECT TO_CHAR(SYSDATE, 'Q') FROM DUAL;          -- 第几季度

-- TO_DATE: 字符串转日期
SELECT TO_DATE('2024-01-01', 'YYYY-MM-DD') FROM DUAL;
SELECT TO_DATE('20240101103000', 'YYYYMMDDHH24MISS') FROM DUAL;

-- TO_TIMESTAMP: 字符串转时间戳
SELECT TO_TIMESTAMP('2024-01-01 10:30:00', 'YYYY-MM-DD HH24:MI:SS') FROM DUAL;

-- 日期示例：计算员工入职年限
SELECT emp_name, hire_date,
       TRUNC(MONTHS_BETWEEN(SYSDATE, hire_date) / 12) AS years
FROM employees;

-- 日期示例：显示上下半年
SELECT emp_name, hire_date,
       CASE 
           WHEN EXTRACT(MONTH FROM hire_date) <= 6 THEN '上半年'
           ELSE '下半年'
       END AS half_year
FROM employees;

NULL处理

核心概念: NULL 表示”未知”或”空值”，它既不是0也不是空字符串。处理NULL需要特殊函数。

-- NVL: 处理单列NULL（两个参数类型必须一致）
SELECT emp_name, NVL(commission_pct, 0) FROM employees;
SELECT NVL('A', 'B') FROM DUAL;  -- 'A'

-- NVL2: 根据是否为NULL返回不同值（3个参数）
-- 语法: NVL2(expr1, expr2, expr3)
-- 如果expr1不为NULL，返回expr2；否则返回expr3
SELECT emp_name, 
       NVL2(commission_pct, '有提成', '无提成') AS commission_status
FROM employees;

-- NULLIF: 比较两个表达式，相等返回NULL，不等返回第一个
-- 语法: NULLIF(expr1, expr2)
SELECT NULLIF(100, 100) FROM DUAL;   -- NULL
SELECT NULLIF(100, 200) FROM DUAL;   -- 100
SELECT NULLIF('A', 'A') FROM DUAL;   -- NULL
SELECT NULLIF('A', 'B') FROM DUAL;   -- A

-- COALESCE: 返回第一个非NULL值（多个参数）
-- 语法: COALESCE(expr1, expr2, expr3, ...)
SELECT COALESCE(NULL, NULL, 'A', 'B') FROM DUAL;  -- A
SELECT COALESCE(NULL, NULL, NULL) FROM DUAL;      -- NULL

-- 实际应用：优先使用手机，否则用固定电话
SELECT emp_name,
       COALESCE(mobile_phone, phone, '无联系方式') AS contact
FROM employees;

-- LNNVL: 处理NULL条件（返回TRUE/FALSE）
-- 用于WHERE子句中，当条件为NULL或FALSE时返回TRUE
SELECT * FROM employees 
WHERE LNNVL(commission_pct > 0.2);  -- 提成<=0.2或NULL的员工

-- NVL/COALESCE 在聚合中的使用
SELECT SUM(NVL(salary, 0)) FROM employees;
SELECT COUNT(NVL(commission_pct, 0)) FROM employees;

-- DECODE处理NULL
SELECT emp_name, 
       DECODE(commission_pct, NULL, '无提成', '有提成') AS status
FROM employees;

通用函数

核心概念: 通用函数是适用于多种数据类型的转换和处理函数。

-- GREATEST: 返回最大值
SELECT GREATEST(10, 5, 20, 15) FROM DUAL;
SELECT GREATEST('A', 'B', 'C') FROM DUAL;
SELECT GREATEST(SYSDATE, SYSDATE + 7, SYSDATE - 7) FROM DUAL;

-- LEAST: 返回最小值
SELECT LEAST(10, 5, 20, 15) FROM DUAL;
SELECT LEAST('A', 'B', 'C') FROM DUAL;

-- SIGN: 返回符号（正数=1，零=0，负数=-1）
SELECT SIGN(100) FROM DUAL;   -- 1
SELECT SIGN(0) FROM DUAL;     -- 0
SELECT SIGN(-100) FROM DUAL;  -- -1
SELECT SIGN(salary - 5000) FROM employees;  -- 判断是否高于5000

-- ABS: 绝对值
SELECT ABS(-100) FROM DUAL;  -- 100

-- FLOOR / CEIL: 向下/向上取整
SELECT FLOOR(3.7) FROM DUAL;  -- 3
SELECT CEIL(3.2) FROM DUAL;   -- 4
SELECT FLOOR(-3.7) FROM DUAL; -- -4
SELECT CEIL(-3.2) FROM DUAL;  -- -3

-- ROUND: 四舍五入
SELECT ROUND(3.456, 2) FROM DUAL;  -- 3.46
SELECT ROUND(3.456, 0) FROM DUAL;  -- 3
SELECT ROUND(3456, -2) FROM DUAL;  -- 3500

-- TRUNC: 截断
SELECT TRUNC(3.456, 2) FROM DUAL;  -- 3.45
SELECT TRUNC(3456, -2) FROM DUAL;   -- 3400
SELECT TRUNC(SYSDATE) FROM DUAL;    -- 截断到天

-- MOD: 取模（余数）
SELECT MOD(10, 3) FROM DUAL;   -- 1
SELECT MOD(10, 2) FROM DUAL;   -- 0
SELECT MOD(salary, 1000) FROM employees;  -- 工资除以1000的余数

-- POWER: 幂运算
SELECT POWER(2, 3) FROM DUAL;   -- 8
SELECT POWER(4, 0.5) FROM DUAL; -- 2 (平方根)

-- SQRT: 平方根
SELECT SQRT(16) FROM DUAL;  -- 4

-- EXP / LN / LOG: 指数和对数
SELECT EXP(1) FROM DUAL;      -- e ≈ 2.718
SELECT LN(2.718) FROM DUAL;   -- 1
SELECT LOG(10, 100) FROM DUAL; -- 2

-- CHR / ASCII: 字符与ASCII码转换
SELECT CHR(65) FROM DUAL;    -- A
SELECT ASCII('A') FROM DUAL; -- 65

-- LENGTH / LENGTHB: 字符长度/字节长度
SELECT LENGTH('你好') FROM DUAL;   -- 2
SELECT LENGTHB('你好') FROM DUAL;  -- 4 (UTF8下一个汉字占3字节)

-- LPAD / RPAD: 左/右填充
SELECT LPAD('ABC', 10, '*') FROM DUAL;  -- *******ABC
SELECT RPAD('ABC', 10, '*') FROM DUAL;  -- ABC*******

-- LTRIM / RTRIM / TRIM: 去除空格或指定字符
SELECT LTRIM('   ABC   ') FROM DUAL;    -- 'ABC   '
SELECT RTRIM('   ABC   ') FROM DUAL;    -- '   ABC'
SELECT TRIM('   ABC   ') FROM DUAL;     -- 'ABC'
SELECT TRIM('A' FROM 'AABCCAA') FROM DUAL;  -- B BCC

-- REPLACE: 替换字符串
SELECT REPLACE('Hello World', 'World', 'Oracle') FROM DUAL;
SELECT REPLACE('A-B-C-D', '-', '') FROM DUAL;

-- TRANSLATE: 字符替换（逐字符映射）
SELECT TRANSLATE('ABC', 'ABC', '123') FROM DUAL;  -- 123
SELECT TRANSLATE('A1B2C3', '12', 'XY') FROM DUAL; -- AXBYC3

-- SUBSTR: 截取子字符串
SELECT SUBSTR('Hello World', 1, 5) FROM DUAL;   -- Hello (从1开始)
SELECT SUBSTR('Hello World', -5) FROM DUAL;      -- World (从右边开始)
SELECT SUBSTR('Hello World', 7, 5) FROM DUAL;    -- World

-- INSTR: 查找子串位置
SELECT INSTR('Hello World', 'o') FROM DUAL;        -- 5 (第一次出现)
SELECT INSTR('Hello World', 'o', 6) FROM DUAL;     -- 8 (从第6位开始找)
SELECT INSTR('Hello World', 'o', 1, 2) FROM DUAL;  -- 8 (第2次出现)

-- CONCAT: 连接字符串（只支持2个参数）
SELECT CONCAT('Hello', 'World') FROM DUAL;  -- HelloWorld

-- || : 字符串连接（多个）
SELECT 'Hello' || ' ' || 'World' FROM DUAL;

-- UPPER / LOWER / INITCAP: 大小写转换
SELECT UPPER('hello') FROM DUAL;    -- HELLO
SELECT LOWER('HELLO') FROM DUAL;    -- hello
SELECT INITCAP('hello world') FROM DUAL;  -- Hello World

-- ROWNUM: 行号（伪列）
SELECT ROWNUM, emp_name FROM employees WHERE ROWNUM <= 10;

-- ROWID: 行地址（伪列，唯一标识每行）
SELECT ROWID, emp_name FROM employees;

DECODE函数和CASE语句

核心概念: DECODE 是 Oracle 特有的函数，用于值的替换；CASE 是标准 SQL 的条件表达式，功能更强大。

-- DECODE: 值的替换（类似switch-case）
-- 语法: DECODE(value, if1, then1, if2, then2, ..., default)

-- 基本用法：简单值替换
SELECT emp_name,
       DECODE(dept_id, 
              10, '财务部',
              20, '市场部',
              30, '研发部',
              '其他部门') AS department
FROM employees;

-- 多值匹配
SELECT emp_name,
       DECODE(job_id,
              'CLERK', '职员',
              'SALESMAN', '销售',
              'MANAGER', '经理',
              '其他') AS job_title
FROM employees;

-- DECODE 配合聚合函数：统计各部门人数
SELECT DECODE(dept_id, 
              10, '财务部',
              20, '市场部',
              30, '研发部',
              '其他') AS dept_name,
       COUNT(*) AS emp_count
FROM employees
GROUP BY DECODE(dept_id, 
                 10, '财务部',
                 20, '市场部',
                 30, '研发部',
                 '其他');

-- CASE 表达式：标准SQL语法，更灵活

-- 简单CASE（类似DECODE）
SELECT emp_name,
       CASE dept_id
           WHEN 10 THEN '财务部'
           WHEN 20 THEN '市场部'
           WHEN 30 THEN '研发部'
           ELSE '其他部门'
       END AS department
FROM employees;

-- 搜索CASE（更强大的条件判断）
SELECT emp_name, salary,
       CASE 
           WHEN salary < 5000 THEN '低薪'
           WHEN salary < 10000 THEN '中等'
           WHEN salary < 20000 THEN '高薪'
           ELSE '超高薪'
       END AS salary_level,
       CASE 
           WHEN hire_date < TO_DATE('2010-01-01', 'YYYY-MM-DD') THEN '老员工'
           WHEN hire_date < TO_DATE('2020-01-01', 'YYYY-MM-DD') THEN '中年员工'
           ELSE '新员工'
       END AS tenure
FROM employees;

-- CASE 在 WHERE 子句中
SELECT * FROM employees
WHERE CASE 
          WHEN salary > 10000 THEN '高收入'
          ELSE '普通收入'
      END = '高收入';

-- CASE 在 ORDER BY 中（动态排序）
SELECT * FROM employees
ORDER BY CASE 
            WHEN job_id = 'PRESIDENT' THEN 1
            WHEN job_id = 'MANAGER' THEN 2
            ELSE 3
         END;

-- CASE 在 UPDATE 中
UPDATE employees
SET salary = CASE 
                WHEN dept_id = 10 THEN salary * 1.1
                WHEN dept_id = 20 THEN salary * 1.15
                ELSE salary * 1.05
             END;

-- CASE 嵌套
SELECT emp_name,
       CASE dept_id
           WHEN 10 THEN 
               CASE job_id
                   WHEN 'MANAGER' THEN '财务经理'
                   ELSE '财务员工'
               END
           WHEN 20 THEN '市场部'
           ELSE '其他'
       END AS detail
FROM employees;

-- CASE 与聚合函数：条件统计
SELECT 
    COUNT(CASE WHEN salary > 10000 THEN 1 END) AS 高薪人数,
    COUNT(CASE WHEN salary BETWEEN 5000 AND 10000 THEN 1 END) AS 中薪人数,
    COUNT(CASE WHEN salary < 5000 THEN 1 END) AS 低薪人数,
    SUM(CASE WHEN dept_id = 10 THEN salary ELSE 0 END) AS 财务部总工资
FROM employees;

-- NULL条件判断
SELECT emp_name,
       CASE 
           WHEN commission_pct IS NULL THEN '无提成'
           WHEN commission_pct < 0.1 THEN '低提成'
           WHEN commission_pct < 0.2 THEN '中提成'
           ELSE '高提成'
       END AS commission_level
FROM employees;

排序函数

核心概念: Oracle 提供强大的分析函数，用于在结果集中进行排序和排名。

-- ROW_NUMBER: 连续排名（无并列），序号唯一不重复
-- 格式：row_number() over(partiton by col1 order by col2)
-- 根据col1分组，在分组内根据col2排序
SELECT emp_name, salary,
       ROW_NUMBER() OVER (PARTITON BY deptno ORDER BY salary DESC) AS rank
FROM employees;

-- RANK: 有并列，跳过排名，数字产生跳跃
SELECT emp_name, salary,
       RANK() OVER (PARTITON BY deptno ORDER BY salary DESC) AS rank
FROM employees;
-- 例如: 10000, 9000, 9000, 8000 -> 排名: 1, 2, 2, 4

-- DENSE_RANK: 有并列，不跳过排名，会产生相同的数字排名
SELECT emp_name, salary,
       DENSE_RANK() OVER (PARTITON BY deptno ORDER BY salary DESC) AS rank
FROM employees;
-- 例如: 10000, 9000, 9000, 8000 -> 排名: 1, 2, 2, 3

-- PARTITION BY: 分组内排名
SELECT dept_id, emp_name, salary,
       ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS dept_rank,
       RANK() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS dept_rank_with_tie
FROM employees;

-- NTILE: 将数据分成N组
SELECT emp_name, salary,
       NTILE(4) OVER (ORDER BY salary DESC) AS quartile  -- 四分位数
FROM employees;

-- LAG / LEAD: 访问前一行/后一行数据
SELECT emp_name, hire_date, salary,
       LAG(salary, 1) OVER (ORDER BY hire_date) AS prev_salary,    -- 前一个人工资
       LEAD(salary, 1) OVER (ORDER BY hire_date) AS next_salary    -- 后一个人工资
FROM employees
WHERE dept_id = 30;

-- LAG/LEAD 带默认值
SELECT emp_name, salary,
       LAG(salary, 1, 0) OVER (ORDER BY salary DESC) AS prev_salary,
       LEAD(salary, 1, 0) OVER (ORDER BY salary DESC) AS next_salary
FROM employees;

-- FIRST_VALUE / LAST_VALUE: 第一/最后一个值
SELECT emp_name, hire_date, salary,
       FIRST_VALUE(salary) OVER (ORDER BY hire_date) AS first_salary,
       LAST_VALUE(salary) OVER (ORDER BY hire_date) AS current_salary
FROM employees;

-- NTH_VALUE: 第N个值
SELECT emp_name, salary,
       NTH_VALUE(salary, 2) OVER (ORDER BY salary DESC) AS second_highest
FROM employees;

-- OVER 完整语法
SELECT emp_name, dept_id, salary,
       ROW_NUMBER() OVER (
           PARTITION BY dept_id    -- 按部门分组
           ORDER BY salary DESC    -- 工资降序
       ) AS dept_rank,
       RANK() OVER (
           ORDER BY salary DESC
       ) AS overall_rank
FROM employees;

-- 多窗口组合
SELECT emp_name, dept_id, salary, hire_date,
       ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS salary_rank,
       ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY hire_date) AS hire_rank
FROM employees;

-- 累计计算
SELECT emp_name, salary,
       SUM(salary) OVER (ORDER BY emp_id) AS cumulative_salary,  -- 累计 sum
       AVG(salary) OVER (ORDER BY emp_id) AS cumulative_avg      -- 累计 avg
FROM employees
WHERE dept_id = 10;

聚合函数

核心概念: 聚合函数对一组值进行计算，返回单个值。常与 GROUP BY 配合使用。

-- 基础聚合函数
SELECT 
    COUNT(*) AS total_count,           -- 总行数
    COUNT(commission_pct) AS has_comm,  -- 非NULL计数
    COUNT(DISTINCT dept_id) AS dept_num,-- 去重计数
    SUM(salary) AS total_salary,        -- 求和
    AVG(salary) AS avg_salary,          -- 平均值
    MAX(salary) AS max_salary,          -- 最大值
    MIN(salary) AS min_salary           -- 最小值
FROM employees;

-- GROUP BY: 分组聚合
SELECT dept_id, job_id,
       COUNT(*) AS emp_count,
       AVG(salary) AS avg_salary,
       SUM(salary) AS total_salary,
       MAX(salary) AS max_salary,
       MIN(salary) AS min_salary
FROM employees
GROUP BY dept_id, job_id
ORDER BY dept_id, job_id;

-- HAVING: 分组后过滤
SELECT dept_id,
       AVG(salary) AS avg_salary
FROM employees
GROUP BY dept_id
HAVING AVG(salary) > 5000;

-- ROLLUP: 聚合小计（层级汇总）
SELECT dept_id, job_id, SUM(salary) AS total_salary
FROM employees
GROUP BY ROLLUP(dept_id, job_id);
-- 结果: (部门+职位)汇总 -> (部门)汇总 -> 总计

-- CUBE: 全量交叉汇总
SELECT dept_id, job_id, SUM(salary) AS total_salary
FROM employees
GROUP BY CUBE(dept_id, job_id);
-- 结果: 所有维度组合的汇总

-- GROUPING: 判断是否小计行
SELECT 
    GROUPING(dept_id) AS is_dept_total,
    GROUPING(job_id) AS is_job_total,
    dept_id, job_id, 
    SUM(salary) AS total_salary
FROM employees
GROUP BY ROLLUP(dept_id, job_id);
-- GROUPING()=1 表示小计行，=0 表示明细行

-- GROUPING SETS: 指定聚合维度
SELECT dept_id, job_id, SUM(salary) AS total_salary
FROM employees
GROUP BY GROUPING SETS(dept_id, job_id);
-- 等同于: GROUP BY dept_id UNION ALL GROUP BY job_id

-- LISTAGG: 字符串聚合（行转列）
SELECT dept_id,
       LISTAGG(emp_name, ',') WITHIN GROUP (ORDER BY emp_name) AS employees
FROM employees
GROUP BY dept_id;

-- 配合 DISTINCT
SELECT dept_id,
       LISTAGG(DISTINCT job_id, ',') WITHIN GROUP (ORDER BY job_id) AS jobs
FROM employees
GROUP BY dept_id;

-- COUNT + CASE 条件统计
SELECT 
    COUNT(CASE WHEN salary > 10000 THEN 1 END) AS 高薪,
    COUNT(CASE WHEN salary BETWEEN 5000 AND 10000 THEN 1 END) AS 中薪,
    COUNT(CASE WHEN salary < 5000 THEN 1 END) AS 低薪
FROM employees;

-- 窗口聚合函数（OVER）
-- 不分组，整个结果集作为窗口
SELECT emp_name, salary,
       SUM(salary) OVER() AS total_salary,
       AVG(salary) OVER() AS avg_salary,
       COUNT(*) OVER() AS total_count
FROM employees;

-- 带 PARTITION BY 的窗口聚合
SELECT dept_id, emp_name, salary,
       SUM(salary) OVER(PARTITION BY dept_id) AS dept_total,
       AVG(salary) OVER(PARTITION BY dept_id) AS dept_avg,
       COUNT(*) OVER(PARTITION BY dept_id) AS dept_count
FROM employees;

-- 滑动窗口（ROWS BETWEEN）
SELECT emp_name, salary,
       SUM(salary) OVER(
           ORDER BY emp_id 
           ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
       ) AS cumulative_sum
FROM employees;
-- UNBOUNDED PRECEDING: 开头
-- CURRENT ROW: 当前行
-- UNBOUNDED FOLLOWING: 结尾

-- 统计函数：标准差、方差
SELECT 
    STDDEV(salary) AS std_dev,     -- 标准差
    VARIANCE(salary) AS variance,  -- 方差
    CORR(salary, commission_pct) AS correlation  -- 相关系数
FROM employees;

-- RATIO_TO_REPORT: 占比
SELECT emp_name, salary,
       ROUND(RATIO_TO_REPORT(salary) OVER() * 100, 2) AS percentage
FROM employees
WHERE dept_id = 10;

-- KEEP (DENSE_RANK FIRST/LAST): 取排序后第一个/最后一个
SELECT dept_id,
       MAX(salary) KEEP (DENSE_RANK FIRST ORDER BY hire_date) AS earliest_hire_salary,
       MIN(salary) KEEP (DENSE_RANK LAST ORDER BY hire_date) AS latest_hire_salary
FROM employees
GROUP BY dept_id;

表关联查询

核心概念: 把多张表拼在一起查，Oracle 支持 ANSI 语法和传统语法。

-- 内连接（INNER JOIN）：两边都有的才显示
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- 左连接（LEFT JOIN）：左边全保留
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id;

-- 右连接（RIGHT JOIN）：右边全保留
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id;

-- 全外连接（FULL OUTER JOIN）：两边都保留
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
FULL OUTER JOIN departments d ON e.dept_id = d.dept_id;

-- 笛卡尔积（CROSS JOIN）：所有组合
SELECT e.emp_name, d.dept_name
FROM employees e
CROSS JOIN departments d;

-- 多表关联
SELECT e.emp_name, d.dept_name, l.city
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
JOIN locations l ON d.location_id = l.location_id;

-- 自连接：自己连自己（上下级关系）
SELECT e.emp_name AS 员工, m.emp_name AS 主管
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.emp_id;

-- 集合运算（ UNION 并集、INTERSECT 交集、MINUS 差集）
SELECT emp_name FROM employees_a
UNION
SELECT emp_name FROM employees_b;

异常处理

核心概念: 异常是程序执行过程中发生的错误，PL/SQL 提供了完善的异常处理机制，就像”错误捕手”，能捕获并处理各种错误情况。

异常类型

-- 异常类型说明：
-- 1. 预定义异常：Oracle 自动定义好的常见错误（如除零、找不到数据等）
-- 2. 非预定义异常：Oracle 错误码对应的异常（需要 PRAGMA EXCEPTION_INIT 关联）
-- 3. 用户自定义异常：业务逻辑中主动定义的异常

-- 1. 预定义异常示例

DECLARE
    v_result NUMBER;
BEGIN
    -- 尝试除以零（会触发预定义异常）
    v_result := 10 / 0;
EXCEPTION
    WHEN ZERO_DIVIDE THEN
        DBMS_OUTPUT.PUT_LINE('不能除以零！');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('其他错误: ' || SQLERRM);
END;
/

-- 常用预定义异常
-- ZERO_DIVIDE       除零 :错误
-- TOO_MANY_ROWS     : SELECT INTO 返回多行
-- NO_DATA_FOUND     : SELECT INTO 没有返回数据
-- INVALID_NUMBER    : 字符串转数字失败
-- VALUE_ERROR       : 类型转换或赋值错误
-- DUP_VAL_ON_INDEX  : 违反唯一索引
-- CURSOR_ALREADY_OPEN: 游标已打开
-- INVALID_CURSOR    : 游标操作无效
-- NOT_LOGGED_ON     : 未连接到数据库
-- TIMEOUT_ON_RESOURCE: 资源等待超时

-- 2. 非预定义异常（将 Oracle 错误码转为异常）

DECLARE
    -- 声明异常
    e_foreign_key EXCEPTION;
    -- 关联错误码
    PRAGMA EXCEPTION_INIT(e_foreign_key, -2292);  -- 违反外键约束
BEGIN
    DELETE FROM departments WHERE dept_id = 10;
EXCEPTION
    WHEN e_foreign_key THEN
        DBMS_OUTPUT.PUT_LINE('无法删除，该部门有员工！');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('错误: ' || SQLERRM);
END;
/

-- 常用 Oracle 错误码
-- -1      : 违反唯一约束
-- -2291   : 违反外键约束（子表有数据）
-- -2292   : 违反外键约束（父表被引用）
-- -1400   : 违反非空约束
-- -12899  : 值太大
-- -1722   : 无效数字
-- -1012   : 未登录

-- 3. 用户自定义异常

DECLARE
    -- 定义异常
    e_salary_low EXCEPTION;
    e_salary_high EXCEPTION;
    -- 定义异常初始化
    PRAGMA EXCEPTION_INIT(e_salary_low, -20001);
    PRAGMA EXCEPTION_INIT(e_salary_high, -20002);
    
    v_salary NUMBER := 1500;
BEGIN
    IF v_salary < 2000 THEN
        -- 抛出异常
        RAISE_APPLICATION_ERROR(-20001, '工资太低了！');
    ELSIF v_salary > 50000 THEN
        RAISE_APPLICATION_ERROR(-20002, '工资太高了！');
    ELSE
        DBMS_OUTPUT.PUT_LINE('工资正常: ' || v_salary);
    END IF;
EXCEPTION
    WHEN e_salary_low THEN
        DBMS_OUTPUT.PUT_LINE('错误: ' || SQLERRM);
    WHEN e_salary_high THEN
        DBMS_OUTPUT.PUT_LINE('错误: ' || SQLERRM);
END;
/

CREATE OR REPLACE PROCEDURE update_salary(

触发器

核心概念: 触发器是”自动触发”的代码，当发生特定事件时自动执行（如增删改）。

-- 创建触发器：插入数据后自动记录
CREATE OR REPLACE TRIGGER trg_emp_insert
AFTER INSERT ON employees
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('新员工已添加: ' || :NEW.emp_name);
END;
/

-- 创建触发器：更新前记录旧值
CREATE OR REPLACE TRIGGER trg_emp_update
BEFORE UPDATE ON employees
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('修改前: ' || :OLD.salary || ', 修改后: ' || :NEW.salary);
END;
/

-- 创建触发器：删除前阻止或记录
CREATE OR REPLACE TRIGGER trg_emp_delete
BEFORE DELETE ON employees
FOR EACH ROW
BEGIN
    RAISE_APPLICATION_ERROR(-20001, '不允许删除员工数据');
END;
/

-- 创建触发器：复合触发器（多个时机）
CREATE OR REPLACE TRIGGER trg_emp_audit
FOR INSERT OR UPDATE OR DELETE ON employees
COMPOUND TRIGGER
    BEFORE STATEMENT IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('操作即将开始');
    END BEFORE STATEMENT;
    
    AFTER EACH ROW IS
    BEGIN
        IF INSERTING THEN
            DBMS_OUTPUT.PUT_LINE('插入: ' || :NEW.emp_name);
        ELSIF UPDATING THEN
            DBMS_OUTPUT.PUT_LINE('更新: ' || :OLD.emp_name || ' -> ' || :NEW.emp_name);
        ELSIF DELETING THEN
            DBMS_OUTPUT.PUT_LINE('删除: ' || :OLD.emp_name);
        END IF;
    END AFTER EACH ROW;
    
    AFTER STATEMENT IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('操作已完成');
    END AFTER STATEMENT;
END trg_emp_audit;
/

-- 禁用/启用触发器
ALTER TRIGGER trg_emp_insert DISABLE;
ALTER TRIGGER trg_emp_insert ENABLE;

-- 删除触发器
DROP TRIGGER trg_emp_insert;

游标

游标就是指向查询结果的指针，游标指向哪条记录，提取的就是哪条记录的数据。

游标的使用流程：

(1)声明游标：

cursor cursorname is s
elect statement;
例如：cursor curemp is
select * from emp;

(2）打开游标：

open cursor_name; 
例如: open cur emp;

(3）从游标中提取数据： 

fetch cursor name into variable[..];
例如：fetch curemp into
v empno,v ename,v job,v mgrv hiredate,v sal,v comm,v deptno;

(4)关闭游标：close cursorname;

核心概念: 游标用于逐行处理查询结果，就像”遍历”数据。

-- 显式游标（手动控制）
DECLARE
    CURSOR emp_cursor IS  -- 1.声明游标
    SELECT emp_id, emp_name, salary FROM employees WHERE dept_id = 10;
    
    v_emp_id NUMBER;
    v_emp_name VARCHAR2(100);
    v_salary NUMBER;
BEGIN
    OPEN emp_cursor;   -- 2.打开游标
    LOOP
        FETCH emp_cursor INTO v_emp_id, v_emp_name, v_salary;   -- 3.从游标提取数据
        EXIT WHEN emp_cursor%NOTFOUND;  -- 没数据了退出
        
        DBMS_OUTPUT.PUT_LINE(v_emp_id || ' - ' || v_emp_name || ' - ' || v_salary);
    END LOOP;
    CLOSE emp_cursor;   --4.关闭游标
END;
/

-- 简化写法：FOR 循环自动开关游标
DECLARE
    CURSOR emp_cursor IS 
        SELECT emp_id, emp_name, salary FROM employees WHERE dept_id = 10;
BEGIN
    FOR emp_rec IN emp_cursor LOOP
        DBMS_OUTPUT.PUT_LINE(emp_rec.emp_id || ' - ' || emp_rec.emp_name);
    END LOOP;
END;
/

-- 更简：直接用子查询
BEGIN
    FOR emp_rec IN (SELECT emp_id, emp_name FROM employees) LOOP
        DBMS_OUTPUT.PUT_LINE(emp_rec.emp_name);
    END LOOP;
END;
/

-- 游标属性
-- emp_cursor%FOUND   -- 是否找到数据，返回TRUE或FALSE
-- emp_cursor%NOTFOUND -- 取反
-- emp_cursor%ROWCOUNT -- 已取出行数
-- emp_cursor%ISOPEN   -- 是否打开

-- 参数化游标
-- cursor 游标名(参数列表) is 查询语句;
DECLARE
    CURSOR emp_cursor(p_dept_id NUMBER) IS 
        SELECT emp_name, salary FROM employees WHERE dept_id = p_dept_id;
BEGIN
    FOR emp_rec IN emp_cursor(10) LOOP
        DBMS_OUTPUT.PUT_LINE(emp_rec.emp_name);
    END LOOP;
END;
/

-- 游标 FOR 循环（简化写法）,for后的变量叫隐藏变量，从in后的语句查询数据赋值给隐藏变量
-- 游标for循环也可以带参数
     FOR emp_record IN (SELECT * FROM employees) LOOP
         DBMS_OUTPUT.PUT_LINE(emp_record.emp_name);
     END LOOP;
    

-- 隐式游标（自动管理，用于单行操作），用于添加删除和修改
BEGIN
    UPDATE employees SET salary = salary + 100 WHERE emp_id = 1;
    IF SQL%FOUND THEN
        DBMS_OUTPUT.PUT_LINE('更新了 ' || SQL%ROWCOUNT || ' 行');
    END IF;
END;
/

事务与锁

核心概念: 事务保证数据一致性，锁防止并发冲突。Oracle 默认隔离级别是 READ COMMITTED。

-- 事务控制
BEGIN
    UPDATE accounts SET balance = balance - 1000 WHERE account_id = 1;
    UPDATE accounts SET balance = balance + 1000 WHERE account_id = 2;
    
    -- 手动提交
    COMMIT;
    
    -- 或者回滚
    -- ROLLBACK;
END;
/

-- 保存点
BEGIN
    INSERT INTO employees VALUES(1, 'A');
    SAVEPOINT sp1;
    INSERT INTO employees VALUES(2, 'B');
    ROLLBACK TO sp1;  -- 回滚到保存点，只保留 A
    COMMIT;
END;
/

-- 行级锁：FOR UPDATE
SELECT * FROM employees WHERE emp_id = 1 FOR UPDATE;

-- 锁住多行
SELECT * FROM employees WHERE dept_id = 10 FOR UPDATE;

-- 加锁超时设置
ALTER SESSION SET LOCK_TIMEOUT = 30;  -- 30秒

其他对象

核心概念: 序列、视图、物化视图、索引、包等是 Oracle 的重要对象。

-- 序列（类似自增，Oracle 没有 AUTO_INCREMENT）
-- 用于生成唯一数字值的数据库对象，生成主键时用
CREATE SEQUENCE seq_emp_id
START WITH 1 
INCREMENT BY 1   -- 设置步长
NOMAXVALUE -- 或 MAXVALUE m 或 NOMINVALUE 或 MINVALUE n
CYCLE  -- NOCYCLE，表示递增到最大或最小之后是否可以重复序列
CACHE 20;   -- 表示在数据库缓存里存储多少个序列的值

-- 使用序列
-- nextval和currval,下一个值和当前值
INSERT INTO employees (emp_id, emp_name) VALUES (seq_emp_id.NEXTVAL, '张三');
SELECT seq_emp_id.CURRVAL FROM DUAL;  -- 获取当前值

-- 删除序列
DROP SEQUENCE seq_emp_id;

-- 视图（虚拟表，不存储数据），简化复杂查询，提供权限访问
-- 简单视图：只从一个表获得且不包含函数运算，分组函数，表达式，剩下的都是复杂视图
-- 下面是复杂视图示例，复杂视图不能进行增删改操作
-- 结尾加上 with read only 就变为只读视图
CREATE OR REPLACE VIEW v_emp_dept AS
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id
WITH READ ONLY;

-- 查询视图
SELECT * FROM v_emp_dept;

-- 物化视图（存储数据，可定时刷新）
CREATE MATERIALIZED VIEW mv_emp_dept
BUILD IMMEDIATE
REFRESH COMPLETE
START WITH SYSDATE NEXT SYSDATE + 1  -- 每天刷新
AS
SELECT e.emp_id, e.emp_name, d.dept_name
FROM employees e
JOIN departments d ON e.dept_id = d.dept_id;

-- 索引（提升查询速度），查询语句中不用指定使用哪个索引
-- 聚集索引：索引顺序和表记录的顺序一致，相对的是非聚集索引
-- 普通索引
CREATE INDEX idx_emp_name ON employees(emp_name);
-- 使用索引
select * from emloyees where emp_name = 'KING';

-- 唯一索引
CREATE UNIQUE INDEX idx_emp_email ON employees(email);

-- 复合索引（最左前缀）
CREATE INDEX idx_emp_dept_sal ON employees(dept_id, salary);

-- 函数索引（基于函数的索引）
CREATE INDEX idx_emp_upper ON employees(UPPER(emp_name));

-- 删除索引
DROP INDEX idx_emp_name;

-- 查看执行计划
EXPLAIN PLAN FOR SELECT * FROM employees WHERE emp_name = '张三';
SELECT * FROM TABLE(DBMS_XPLAN.DISPLAY);

-- 同义词（对象的别名）
CREATE SYNONYM emp FOR employees;
SELECT * FROM emp;  -- 等于 SELECT * FROM employees

-- 包（封装过程、函数、变量），类似于JAVA中的类
-- 包由两部分组成：包定义和包主体
-- 创建包规范（定义）
CREATE OR REPLACE PACKAGE emp_pkg IS
    PROCEDURE update_salary(p_emp_id NUMBER, p_salary NUMBER);
    FUNCTION get_emp_name(p_emp_id NUMBER) RETURN VARCHAR2;
END emp_pkg;
/

-- 创建包体
CREATE OR REPLACE PACKAGE BODY emp_pkg IS
    PROCEDURE update_salary(p_emp_id NUMBER, p_salary NUMBER) IS
    BEGIN
        UPDATE employees SET salary = p_salary WHERE emp_id = p_emp_id;
    END;
    
    FUNCTION get_emp_name(p_emp_id NUMBER) RETURN VARCHAR2 IS
        v_name VARCHAR2(100);
    BEGIN
        SELECT emp_name INTO v_name FROM employees WHERE emp_id = p_emp_id;
        RETURN v_name;
    END;
END emp_pkg;
/

-- 调用包
BEGIN
    emp_pkg.update_salary(1, 10000);
    DBMS_OUTPUT.PUT_LINE(emp_pkg.get_emp_name(1));
END;
/

常用系统表/视图

核心概念: Oracle 提供大量数据字典视图查询元数据。

-- 查询用户下的所有表
SELECT table_name FROM user_tables;

-- 查询表的所有列
SELECT column_name, data_type, data_length, nullable
FROM user_tab_columns WHERE table_name = 'EMPLOYEES';

-- 查询表的约束
SELECT constraint_name, constraint_type, column_name
FROM user_cons_columns 
JOIN user_constraints USING (constraint_name)
WHERE table_name = 'EMPLOYEES';

-- 查询索引
SELECT index_name, column_name
FROM user_ind_columns WHERE table_name = 'EMPLOYEES';

-- 查询序列
SELECT sequence_name, last_number FROM user_sequences;

-- 查询视图定义
SELECT text FROM user_views WHERE view_name = 'V_EMP_DEPT';

-- 查询当前会话信息
SELECT user, SYSDATE, SYSTIMESTAMP FROM DUAL;

-- 查询 Oracle 版本
SELECT * FROM v$version;
